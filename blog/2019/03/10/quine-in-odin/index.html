2019-03-10

A Quine in Odin:

package quine

import "core:fmt"

main :: proc() {
	fmt.printf("%s%c%s%c;\n", s, 0x60, s, 0x60);
}

s := `package quine

import "core:fmt"

main :: proc() {
	fmt.printf("%s%c%s%c;\n", s, 0x60, s, 0x60);
}

s := `;
